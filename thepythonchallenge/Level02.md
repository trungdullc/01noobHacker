from urllib.request import urlopen

# with urlopen("http://www.pythonchallenge.com/pc/def/ocr.html") as page:
page = urlopen("http://www.pythonchallenge.com/pc/def/ocr.html")
contents = page.read().decode('utf-8')
page.close()

# print(contents.split("-->")[1].split("<--\n")[0])
mess = contents.split("-->")[1].split("<--\n")[0]

# Empty dictionary
items = {}

# check if item is in dictionary and count
for item in mess:
    if item in items:
        items[item] += 1
    else:
        items[item] = 1

found = []

# Note: Python2 has .iterkeys() not Python3
# for key in items.iterkeys():
for key in items:
    if items[key] > 1:
        found += [key]              # ['\n', '!', '-', '%', '$', '@', '_', '^', '#', ')', '&', '+', ']', '*', '}', '[', '(', '{']

for item in found:
    mess = mess.replace(item, "")   # rm

print(mess)
