# Side Quest: Coffee Machine
```python
LOGO = """
_________         _____  _____                   
\_   ___ \  _____/ ____\/ ____\____   ____       
/    \  \/ /  _ \   __\\   __\/ __ \_/ __ \      
\     \___(  <_> )  |   |  | \  ___/\  ___/

        \/                        \/     \/      
   _____                .__    .__               
  /     \ _____    ____ |  |__ |__| ____   ____  
 /  \ /  \\__  \ _/ ___\|  |  \|  |/    \_/ __ \ 
/    Y    \/ __ \\  \___|   Y  \  |   |  \  ___/ 
\____|__  (____  /\___  >___|  /__|___|  /\___  >
        \/     \/     \/     \/        \/     \/ 
"""

# Menu options and settings
MENU = {
    "espresso": {
        "ingredients": {
            "water": 50,
            "coffee": 18,
        },
        "cost": 1.5,
    },
    "latte": {
        "ingredients": {
            "water": 200,
            "milk": 150,
            "coffee": 24,
        },
        "cost": 2.5,
    },
    "cappuccino": {
        "ingredients": {
            "water": 250,
            "milk": 100,
            "coffee": 24,
        },
        "cost": 3.0,
    }
}

# Machine starting values
resources = {
    "water": 300,
    "milk": 200,
    "coffee": 100,
    "money": 0.00
}

# Initialize global variables on machine
off = False                             # similar to game_over
money = 0.00                            # global variable money is total money in machine

def prompt() -> str:
    """
    Prompt user espresso/latte/cappuccino
    also used for report and off commands
    """
    return input("What would you like? (espresso/latte/cappuccino): ").lower()[0]

def check_input(input, money):
    req_water = MENU[input]['ingredients']['water']
    req_coffee = MENU[input]['ingredients']['coffee']
    if 'milk' in MENU[input]['ingredients']:                    # Note: Safety if 'milk' in MENU
        req_milk = MENU[input]['ingredients']['milk']
    req_money = float(MENU[input]['cost'])

    if money >= req_money:                                      # If enough money check if enough ingredients
        if req_water > resources['water']:
            print("Not enough water. Make another selection.")
        elif req_coffee > resources['coffee']:
            print("Not enough coffee. Make another selection.")
        elif 'milk' in MENU[input]['ingredients'] and req_milk > resources['milk']:
            print("Not enough milk. Make another selection")
        else:
            # Remove machine resources per order
            resources['water'] -= req_water
            resources['coffee'] -= req_coffee

            if 'milk' in MENU[input]['ingredients']:
                resources['milk'] -= req_milk

            # Refund surplus money to customer, add required amount to resources
            if money > req_money:
                print(f"Here is ${money - req_money:.2f} in change.")

            resources['money'] += float(req_money)
            print(f"Enjoy your {input}!")
    else:
        print("Sorry that's not enough money. Money refunded")

def get_coins():
    """
    Calculate coin total
    """
    quarters: float = float(input("How many quarters: "))
    dimes: float = float(input("How many dimes: "))
    nickels: float = float(input("How many nickels: "))
    pennies: float = float(input("How many pennies: "))

    return (quarters * .25) + (dimes * .10) + (nickels * .05) + (pennies * .01)

if __name__ == "__main__":
    print(LOGO)

    while not off:
        user_input = prompt()

        if user_input == "o":
            off = True
        elif user_input == "r":
            print(f"\nWater: {resources['water']}ml")
            print(f"Milk: {resources['milk']}ml")
            print(f"Coffee: {resources['coffee']}g")
            print(f"Money: ${resources['money']:.2f}\n")
        elif user_input == "e":
            check_input("espresso", get_coins())
        elif user_input == "l":
            check_input("latte", get_coins())
        elif user_input == "c":
            check_input("cappuccino", get_coins())
        else:
            print("Invalid request. Try again.")
```
