# Side Quest: Control snake movement
```python
from turtle import Turtle, Screen
import time

STARTING_POSITIONS = [(0, 0), (-20, 0), (-40, 0)]
MOVE_DISTANCE = 20
UP = 90
DOWN = 270
LEFT = 180
RIGHT = 0

class Snake:
    def __init__(self):
        self.segments = []
        self.create_snake()
        self.head = self.segments[0]                        # Note: if use alot turn into a attribute
    
    def create_snake(self):
        for position in STARTING_POSITIONS:
            new_segment = Turtle(shape = "square")
            new_segment.penup()
            new_segment.color("green")
            new_segment.setposition(position)
            self.segments.append(new_segment)

    def move(self):
        for seg_index in range(len(self.segments) - 1, 0, -1):
            new_x = self.segments[seg_index - 1].xcor()
            new_y = self.segments[seg_index - 1].ycor()
            self.segments[seg_index].setposition(new_x, new_y)

        self.segments[0].fd(MOVE_DISTANCE)                  # snake.head.fd(MOVE_DISTANCE)

    def up(self):                                           # New
        if self.head.heading() != DOWN:
            # self.segments[0].setheading(90)
            snake.head.setheading(UP)

    def down(self):
        if self.head.heading() != UP:
            # self.segments[0].setheading(270)
            snake.head.setheading(DOWN)

    def left(self):
        if self.head.heading() != RIGHT:
            # self.segments[0].setheading(180)
            snake.head.setheading(LEFT)

    def right(self):
        if self.head.heading() != LEFT:
            # self.segments[0].setheading(0)
            snake.head.setheading(RIGHT)

gameover = False

if __name__ == "__main__":
    screen = Screen()
    screen.setup(width = 600, height = 600)
    screen.bgcolor("black")
    screen.title("Snake Game")
    screen.tracer(0)

    snake = Snake()
    screen.listen()                                         # New
    screen.onkey(snake.up, "Up")
    screen.onkey(snake.down, "Down")
    screen.onkey(snake.left, "Left")
    screen.onkey(snake.right, "Right")

    while not gameover:
        screen.update()
        time.sleep(0.1)
        snake.move()

    screen.exitonclick()
```
