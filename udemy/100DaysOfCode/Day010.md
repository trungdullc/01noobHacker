        result = first_number * second_number

# Side Quest: Leap year
```python
# Google: leap year math
# Normal year = 365 days
# Leap year = 366 days with extra day in Feb
# To determine if a year is a leap year, follow these rules:
#   if divisible by 4 == leap year
#       if divisible by 100 == not leap year
#           if divisible by 400 == leap year

def is_leap_year(year: int) -> bool:
    if year % 4 == 0:
        if year % 100 == 0:
            if year % 400 == 0:
                return True
            return False
        return True

if __name__ == "__main__":
    print(is_leap_year(400))
```
