# Python Built-in Functions
# Documentation: https://docs.python.org/3/library/functions.html

PEP 8: https://peps.python.org/pep-0008/

global_counter = 0

def counter() -> bool:
    global global_counter                   # Note: Need global in python

    if global_counter != 3:
        global_counter += 1
        return False
    return True

while counter() != True:                    # while not counter()
    print("Hi")
