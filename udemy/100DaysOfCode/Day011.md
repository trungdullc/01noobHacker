# TODO-1: Create a deal_card() fx that uses the dict cards to return a random card
def deal_card(cards) -> int:
    """
    Returns a random card back
    """
    return random.choice(list(cards.values()))

def calculate_score(cards) -> int:
    """
    Removes 11 and appends 1 to follow the rules, then uses builtins.sum()
    """
    if sum(cards) == 21 and len(cards) == 2:
        return 0                                        # Special for Black Jack
    elif 11 in cards and sum(cards) > 21:
        cards.remove(11)
        cards.append(1)
    return sum(cards)

if __name__ == "__main__":

        for _ in range(2):
            user_cards.append(deal_card(cards))
            computer_cards.append(deal_card(cards))

        # print(user_cards)

        print(f"Your cards: {user_cards}, current score: {calculate_score(user_cards)}")
        print(f"Computer's first card: {computer_cards[0]}")

        deal: str = 'y'

        while deal == "y" and calculate_score(user_cards) <= 21:
            deal: str = input("Type 'y' to get another card, type 'n' to pass: ").lower()[0]

                user_cards.append(deal_card(cards))
                print(f"Your cards: {user_cards}, current score: {calculate_score(user_cards)}")

                ...

        # Force computer to keep picking
        while sum(computer_cards) < 17:
            computer_cards.append(deal_card(cards))

        # Deciding winner

        elif calculate_score(user_cards) > 21:                                  # Player Bust

            print("Computer bust. You win")
