# Side Quest: Guess the number
```python
import random

logo = """
  / _ \\_   _  ___  ___ ___  /__   \\ |__   ___    /\\ \\ \\_   _ _ __ ___ | |__   ___ _ __ 
 / /_\\/ | | |/ _ \\/ __/ __|   / /\\/ '_ \\ / _ \\  /  \\/ / | | | '_ ' _ \\| '_ \\ / _ \\ '__|
/ /_\\\\| |_| |  __/\\__ \\__ \\  / /  | | | |  __/ / /\\  /| |_| | | | | | | |_) |  __/ |   
\\____/ \\__,_|\\___||___/___/  \\/   |_| |_|\\___| \\_\\ \\/  \\__,_|_| |_| |_|_.__/ \\___|_|   
"""

correct_answer = random.choice(range(1,100))
guessed = False
attempts = 0

if __name__ == "__main__":
    print(logo)

    print("\nWelcome to the Number Guessing Game!")
    print(f"DEBUG: Psst, the correct answer is {correct_answer}")

    print("I'm thinking of a number between 1 and 100.")
    difficulty: str = input("Choose a difficulty. Type 'easy' or 'hard': ").lower()[0]

    if difficulty == "e":
        attempts = 10
    else:
        attempts = 5

    while not guessed and attempts > 0:
        print(f"You have {attempts} attempts remaining to guess the number.")
        my_guess: int = int(input("Make a guess: "))
        
        if my_guess == correct_answer:
            print("You win")
            guessed = True
        elif my_guess > correct_answer:
            print("To high.\nGuess again.")
            attempts -= 1
        else:
            print("To low.\nGuess again")
            attempts -= 1
    
    if attempts == 0:
        print("Sorry you ran out of attempts")
```

# Side Quest: global and local scope
```python
# Note: Different than C++/Java, need global keyword
enemies: int = 1

def increase_enemies():
    enemies: int = 2                                        # local scope, Note: NameError if try access outside scope
    print(f"enemies inside increase_enemies() fx: {enemies}")

def increase_enemies4():
    global enemies                                          # global scope, Note: Good practice not to modify global variables
    enemies = 4
    print(f"enemies inside increase_enemies4() fx: {enemies}")

if __name__ == "__main__":
    increase_enemies()
    print(f"enemies outside fx: {enemies}")                 # enemies outside fx: 1
    increase_enemies4()
    print(f"enemies outside fx: {enemies}")                 # enemies outside fx: 4
```

# Side Quest: Prime Checker
```python
# Prime numbers Checker
# Divided by themselves and 1 = True

def is_prime(num: int) -> bool:
    """
    Check if number argument is prime or not and returns True/False
    """
    prime_checker: list[int] = []

    for i in range(1, num + 1):
        if num % i == 0:
            prime_checker.append(i)
    
    # print(prime_checker)                      # DEBUG algorithm
    return len(prime_checker) == 2

if __name__ == "__main__":
    print(is_prime(7))                          # True, [1,7]
    print(is_prime(4))                          # False, [1, 2, 4]
    print(is_prime(73))                         # True
```
