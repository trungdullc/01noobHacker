# Side Quest: Etch-a-Sketch 
```python
# w: forward
# s: backwards
# a: counter-clockwise
# d: clockwise
# c: clear drawing

from turtle import Turtle, Screen

def move_forwards():
    turtle.fd(10)

def move_backwards():
    turtle.backward(10)

def turn_left():
    new_heading = turtle.heading() + 10             # turtle.left(10)
    turtle.setheading(new_heading)

def turn_right():
    new_heading = turtle.heading() - 10             # turtle.right(10)
    turtle.setheading(new_heading)

def clear_screen():
    turtle.clear()
    turtle.penup()
    turtle.home()                           # original position (0,0)
    turtle.pendown()

if __name__ == "__main__":
    turtle = Turtle()
    turtle.shape("turtle")
    turtle.color("red", "green")

    screen = Screen()

    screen.listen()
    screen.onkey(move_forwards, "w")
    screen.onkey(move_backwards, "s")
    screen.onkey(turn_left, "a")
    screen.onkey(turn_right, "d")
    screen.onkey(clear_screen, "c")
    screen.exitonclick()
```

# Side Quest: 5 color turtle racing game
```python
from turtle import Turtle, Screen
import random

""" width: 500, height: 400

            ^ 200
            |
            |            
-250        |           250
<---------(0,0)--------->
            |
            |            
            |
            v -200

Default turtle is 20 x 20
"""

def move_forwards(turtle):
    turtle.fd(10)

turtle_colors = ["red", "green", "blue", "yellow", "purple"]
y_positions = [-70, -40, -10, 20, 50]
all_turtles = []
is_race_on = False

if __name__ == "__main__":
    for turtle_index in range(len(turtle_colors)):
        turtle = Turtle(shape = "turtle")                               # turtle is basically a new_turtle
        turtle.color(turtle_colors[turtle_index])
        turtle.penup()
        turtle.setposition(x = -230, y = y_positions[turtle_index])
        all_turtles.append(turtle)                                      # append so can be deleted

    screen = Screen()
    screen.setup(width = 500, height = 400)
    user_bet = screen.textinput(title = "Make your bet", prompt = "Enter a color [red, green, blue, yellow, purple]: ").lower()[0]
    print(user_bet)

    if user_bet:
        is_race_on = True

    while is_race_on:
        for turtle in all_turtles:
            if turtle.xcor() > 230:
                is_race_on = False
                winning_color = turtle.pencolor().lower()[0]
                if winning_color == user_bet:
                    print("You picked the winning turtle!")
                else:
                    print("You lost")
            random_distance = random.randint(0, 10)
            turtle.fd(random_distance)

    screen.exitonclick()
```
