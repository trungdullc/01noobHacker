import random

# if word_list in hangman_words.py

correct_letters: list[str] = []

# if stages in hangman_art.py

logo = """

                    __/ |                      
                   |___/
"""

# TODO-9 - Create a variable called 'lives' to keep track of the number of lives left. Set it to 6
lives: int = 6

print(logo)

# TODO-1 - Randomly choose a word from word_list and assign it to a variable called chosen_word
chosen_word: str= random.choice(word_list)
print(f"DEBUG: chosen_word: {chosen_word}")

# TODO-4 - Create a "placeholder" with same number of blanks as chosen_word
placeholder: str = len(chosen_word) * "-"
print(placeholder)

# TODO-6 - Use a while loop to let user guess again
game_over = False

while not game_over:
    # TODO-13 - Updated lives
    print(f"*************** {lives}/6 LIVES LEFT ***************")
    
    # TODO-2 - Ask user to guess a letter and assign it to guess.  Make guess lowercase
    guess: str = input("Guess a letter: \n").lower()
    # print(guess[0])

    # TODO-12 - If user has entered a letter they've already guessed, print "You've already guessed"
    if guess in correct_letters:
        print(f"You've already guessed {guess}")

    # TODO-3 - Check if the letter user guessed (guess) is one of the letters in the chosen_word.  Print "Right" if it is, "Wrong" if it's not
    # for char in chosen_word:
    #     if guess[0] == char:
    #         print("Right")
    #     else:
    #         print("Wrong")

    # TODO-5 - Create a "display" that puts theguess letter in the right positions else _
    display: str = ""

    for char in chosen_word:
        if guess[0] == char:
            display += char
            # TODO-8 - Change for loop so that keep previous correct letters in display
            correct_letters.append(char)
        elif char in correct_letters:
            display += char
        else:
            display += "_"

    print(display)

    # TODO-10 - If guess not a letter in chosen_word.  Then reduce 'lives" by 1 and if lives goes to 0, then print "You Lose"
    if guess not in chosen_word:
        lives -= 1
        if lives == 0:
            game_over = True
            print("You Lose")

    # TODO-7 - End game if no more _ in display
    if "_" not in display:
        game_over = True
        print("You Win")
    
    # TODO-11 - print ASCII art from "stages" that corresponds to current number of "lives"
    print(stages[lives])
