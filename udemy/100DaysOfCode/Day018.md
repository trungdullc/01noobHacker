```python
from turtle import Turtle, Screen

def dotted_line():
    for _ in range(10):
        turtle.forward(10)
        turtle.penup()
        turtle.forward(10)
        turtle.pendown()

if __name__ == "__main__":
```

# Side Quest: turtle draw a spirograph
```python
import random
from turtle import Turtle, Screen, colormode

# 0: E, 90: N, 180: W, 270: S
directions: list[int] = [0, 90, 180, 270]

def random_color() -> tuple[int, int, int]:                 # Note: tuple is immutable
    red = random.randint(0, 255)
    green = random.randint(0, 255)
    blue = random.randint(0, 255)

    return (red, green, blue)

def draw_spirograph(size_of_gap: int):
    for _ in range(int(360 / size_of_gap)):
        turtle.color(random_color())
        current_head = turtle.heading()
        turtle.setheading(current_head + size_of_gap)
        turtle.circle(100)

if __name__ == "__main__":
    turtle = Turtle()
    turtle.shape("turtle")
    turtle.color("red", "green")
    colormode(255)
    turtle.speed("fastest")

    draw_spirograph(5)

    screen = Screen()
    screen.exitonclick()
```

# Side Quest: Use extracted color_list to draw hirst painting
```python
import random
from turtle import Turtle, Screen, colormode

colormode(255)

# Used the color_list
color_list = [(236, 220, 120), (26, 109, 183), (225, 151, 82), (216, 64, 98), (213, 131, 164), (142, 85, 50), (122, 149, 203), (41, 191, 119)]
number_of_dots = 20

if __name__ == "__main__":
    turtle = Turtle()
    turtle.shape("turtle")
    turtle.color("red", "green")
    turtle.hideturtle()
    turtle.speed("fast")
    turtle.penup()                                          # Removes line

    turtle.setheading(225)                                  # Note: Default is 0 - east 0 - north (going to right)
    turtle.fd(250)
    turtle.setheading(0)                                    # reset way turtle faces

    for dot_count in range(1, number_of_dots + 1):
        turtle.dot(20, random.choice(color_list))
        turtle.fd(50)

        if dot_count % 5 == 0:
            turtle.setheading(90)
            turtle.fd(50)
            turtle.setheading(180)
            turtle.fd(250)                                  # move it to start to reuse code
            turtle.setheading(0)


    screen = Screen()
    screen.exitonclick()
```
