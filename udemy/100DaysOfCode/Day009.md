# Side Quest: Implement bidding program from flowchart
```python
logo: str = """
                         ___________
                         \         /
                          )_______(
                          |"""""""|_.-._,.---------.,_.-._
                          |       | | |               | | ''-.
                          |       |_| |_             _| |_..-'
                          |_______| '-'''---------'' '-'
                          )"""""""(
                         /_________\\
                       .-------------.
                      /_______________\\
"""

bidders_list: dict[str, int] = {}

def calculate_winner():
    highest_bid_name: str = ""
    highest_bid: int = 0

    for name in bidders_list:
        if bidders_list[name] > highest_bid:
            highest_bid = bidders_list[name]
            highest_bid_name = name

    print(f"{highest_bid_name} won with bid of ${highest_bid}")

if __name__ == "__main__":
    gameover: bool = False

    print(logo)
    print("Welcome to the secret auction program")

    while not gameover:
        name:str = input("What is your name?: ")
        bid:int  = int(input("What's your bid?: $"))
        bidders_list[name] = bid

        is_finished = input("Are there any other bidders? Type 'yes' or 'no'.\n").lower()[0]

        if is_finished == "n":
            gameover = True
    
    calculate_winner()
```
