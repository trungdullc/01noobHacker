```python
print(5/3)              # Float Division
print(5//3)             # Int Division: Float Division first than truncation
print(5**2)             # power
print(5%3)              # modulus

def power(number: float, power: int) -> float:
    """
    Custom power function without using built-in **
    """
    ans = 1
    for _ in range(power):
        ans *= number
    return ans

print(power(2,4))
```
