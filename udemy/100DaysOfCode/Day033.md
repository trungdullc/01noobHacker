# Side Quest: ISS (International Space Station) Tracker
```python
# When ISS is above my location and currently dark (sunset) then send email to myself to look up
# BONUS: Run code every 60 sec

import requests
from datetime import datetime
import smtplib
import time

MY_LAT = 51.507351
MY_LONG = -0.127758
MY_EMAIL = "duprogramllc@gmail.com"
MY_PASSWORD = "PLACEHOLDER"

parameters = {
    "lat": MY_LAT,
    "lng": MY_LONG,
    "formatted": 0,
}

def is_iss_overhead(iss_latitude, iss_longitude) -> bool:
    if abs(MY_LAT - iss_latitude) < 5 and MY_LONG - 5 <= iss_latitude <= MY_LONG + 5:
        return True
    return False

def is_night() -> bool:
    if time_now >= sunset or time_now <= sunrise:
        return True
    return False

if __name__ == "__main__":
    response = requests.get(url="http://api.open-notify.org/iss-now.json")
    response.raise_for_status()

    data = response.json()

    iss_latitude = float(data["iss_position"]["latitude"])
    iss_longitude = float(data["iss_position"]["longitude"])

    # Your position is within +5 or -5 degrees of the ISS position.
    is_iss_overhead(iss_latitude, iss_longitude)

    response = requests.get("https://api.sunrise-sunset.org/json", params=parameters)
    response.raise_for_status()
    data = response.json()

    sunrise = int(data["results"]["sunrise"].split("T")[1].split(":")[0])
    sunset = int(data["results"]["sunset"].split("T")[1].split(":")[0])

    time_now = datetime.now().hour

    print(is_night())

    # BONUS: run the code every 60 seconds
    while True:
        time.sleep(60)
        # If ISS is close to my current position and it is currently dark
        # send me an email to tell me to look up
        if is_iss_overhead and is_night:
            print("ISS is nearby look up, send email with smtplib")
            with smtplib.SMTP("smtp.gmail.com") as server:
                server.starttls()
                server.login(user=MY_EMAIL, password=MY_PASSWORD)
                server.sendmail(
                    from_addr=MY_EMAIL,
                    to_addrs=MY_EMAIL,
                    msg="Subject: Look up.  The ISS is above you in the sky!"
                )
```
