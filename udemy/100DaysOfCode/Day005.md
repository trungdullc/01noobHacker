# Side Quest: custom sum
```python
from builtins import sum

def custom_sum(mylist: list[int]) -> int:
    """
    Created a custom function similar to builtins.sum()
    """
    sum_counter: int = 0
    for score in mylist:
        sum_counter += score
    return sum_counter

if __name__ == "__main__":
    student_scores: list[int] = [150, 142, 185, 120, 171, 184, 149, 24, 59, 68, 199, 78, 65, 89, 86, 3, 100]

    # Comparing builtins.sum() with custom_sum(), should be same
    print(sum(student_scores))
    print(custom_sum(student_scores))
```

```python
from builtins import max

def custom_max(mylist: list[int]) -> int:
    """
    Created a custom function similar to builtins.max()
    """
    max_number: int = mylist[0]

    for score in mylist[1:]:
        if score > max_number:
            max_number = score
            
    return max_number
```

# Side Quest: Gauss summation formula ⭐⭐⭐⭐⭐
S = $\sum_{i=1}^{n} i$ = $\frac{n(n+1)}{2}$
```python
from builtins import range

if __name__ == "__main__":
    current_total = 0
    
    # Runtime Complexity: O(n)
    # Space Complexity: O(1)
    for i in range(1, 101, 1):
        current_total += i
    
    print(current_total)

    # Gauss Summation
    # Runtime Complexity: O(1)
    # Space Complexity: O(1)
    n = 100
    S: float = n * (n+1) / 2
    print(S)
```

```python
def fizzbizz() -> None:
    """
    Rules: 
        print number 1 to 100(include)
        if number divisible by 3 print 'Fizz'       index%3==0
        if number divisible by 5 print 'Buzz'       index%5==101
        if divisible by 3 and 5 print 'FizzBuzz'    Just make it two seperate if statements
    """
    for index in range(1,101,1):
        if index%3==0:
            if index%5==0:
                print("FizzBuzz")
                continue
            print("Fizz")
        elif index%5==0:
            print("Buzz")
        else:
            print(index)

if __name__ == "__main__":
```

```python
print("Welcome to the PyPassword Generator!")
nr_letters: int = int(input("How many letters would you like in your password?\n")) 
nr_symbols: int = int(input("How many symbols would you like?\n"))
nr_numbers: int = int(input("How many numbers would you like?\n"))

def str_password_generator() -> None:
    """
    Learn random.choice() and random.shuffle()
    """
    password: str = ""

    for char in range(1, nr_letters + 1):
        password += random.choice(letters)

    for char in range(1, nr_symbols + 1):
        password += random.choice(symbols)

    for char in range(1, nr_numbers + 1):
        password += random.choice(numbers)

    password_list = list(password)                      # Fix: convert str to mutable type then back to str
    random.shuffle(password_list)                       # random.shuffle() only works on MutableSequence[Any] but string is immutable
    password = "".join(password_list)                   # convert list back to str
    print(f"Your password is: {password}")

def list_password_generator() -> None:
    """
    start with list so no need to convert from str to list and back to str
    """
    password_list: list[str] = []

    for char in range(1, nr_letters + 1):
        password_list.append(random.choice(letters))

    for char in range(1, nr_symbols + 1):
        password_list += random.choice(symbols)

    for char in range(1, nr_numbers + 1):
        password_list += random.choice(numbers)

    print(password_list)
    random.shuffle(password_list)
    # print again to see how random.shuffle() works
    print(password_list)

    password: str = ""                                           # other way to builtins.join()
    for char in password_list:
        password += char

    print(f"Your password is: {password}")

if __name__ == "__main__":
    str_password_generator()
    list_password_generator()
```
