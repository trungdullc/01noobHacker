# Side Quest: Creating custom module ⭐⭐⭐⭐⭐
```python
# my_module.py
my_fav_number: float = 3.1415

def double(x: float) -> float:
    return x * 2

class MathHelper:
    def __init__(self, value: float) -> None:
        self.value = value

    def double(self) -> float:
        return self.value * 2

# app.py
import my_module

if __name__ == "__main__":
    print(my_module.my_fav_number * 2)
    print(my_module.double(3.1415))

    helper = my_module.MathHelper(3.1415)               # from my_module import MathHelper ❤️
    print(helper.double())
```

# Side Quest: Rock, Paper, Scissor
```python
import random

rock: str = """
    _______
---'   ____)
      (_____)
      (_____)
      (____)
---.__(___)

"""

paper: str = """
    _______
---'   ____)____
          ______)
          _______)
         _______)
---.__________)

"""

scissor: str = """
    _______
---'   ____)____
          ______)
       __________)
      (____)
---.__(___)

"""

game_images: list[str] = [rock, paper, scissor]

# Note: If have long if statement use ()
def game_logic(user_choice: int, computer_choice: int) -> str:
    if user_choice < 0 or user_choice > 2:
        return "Invalid Number. You automatically lose"
    elif ((user_choice == 0 and computer_choice == 2) or                         
          (user_choice == 1 and computer_choice == 0) or 
          (user_choice == 2 and computer_choice ==1)):
        return "You win"
    elif ((user_choice == 0 and computer_choice == 1) or                         
          (user_choice == 1 and computer_choice == 2) or 
          (user_choice == 2 and computer_choice ==0)):
        return "You lose"
    else:
        return "You both tie"
    
if __name__ == "__main__":
    user_choice: int = int(input("[0]: Rock, [1]: Paper, [2]: Scissor\n"))
    computer_choice: int = random.randint(0, 2)

    try:
        print(f"Human choose:\n {game_images[user_choice]}")
        print(f"Computer choose:\n {game_images[computer_choice]}")
        print(game_logic(user_choice, computer_choice))
    except Exception as e:
        print(f"Error: {e}, Please enter valid number next time\n")
    finally:
        print("Have a nice day")
```
