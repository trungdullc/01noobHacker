# Difference between python2 and python3
```
# python2                                   # python3
import Tkinter as tk                        import tkinter as tk                # Reason for alias to work with both
import tkFileDialog as filedialog           from tkinter import filedialog
import ttk                                  from tkinter import ttk

try:
    import Tkinter
except ModuleNotFoundError:
    from tkinter import *

Additional Windows should be Toplevel instances
```
